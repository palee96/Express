for (let i = 1; i <= 100; i++)
    {
        if ((i % 3 || i % 7) ==0){
            console.log("FizzBuzz\n");
        }
       
         else if (i % 3 == 0)
        {
            console.log("Fizz\n");
        }
        else if (i % 7 ==0)
        {
            console.log("Buzz\n");
        }
       
        else{
            console.log("%d\n",i);
        }
    }
